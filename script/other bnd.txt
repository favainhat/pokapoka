idstring "BND"

goto 0xC
get TABLEOFFSET long
get FILES long

#print %TABLEOFFSET%
#print %FILES%

goto 0x20

for i = 0 < FILES
    get INDEX long
    get OFFSET long
    get SIZE long
    get NAMEOFFSET long
    savepos TEMP
    goto NAMEOFFSET
    get NAME string
    log NAME OFFSET SIZE
    goto TEMP
next i
